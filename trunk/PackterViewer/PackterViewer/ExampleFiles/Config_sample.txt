# HtmlConvertTarget
# PACKTERHTML や PACKTERMSG で入力された IEコンポーネント に渡される文字列を、
# 文字列１ 文字列２ とあったら、文字列１ のものを 文字列２ に置き換えます。
# ここの例に示されているものだった場合、
# <img src="<HOGE>">
# は、
# <img src="file:///c:/tmp/img.png">
# と書き換えられます。
# また、HtmlConvertTarget は複数指定できます。
# default は無指定です。
#HtmlConvertTarget = <HOGE> file:///c:/tmp/img.png
#HtmlConvertTarget = <HAGE> file:///c:/tmp/img2.png
HtmlConvertTarget = <pic-m1> <img src="file:///<CURRENTDIR>/pic01.png" height="150" align="top"></td><td width="<HTMLWIDTH>-<HTMLHEIGHT>*2" align="center">
HtmlConvertTarget = <pic-m2> <img src="file:///<CURRENTDIR>/pic02.png" height="150" align="top"></td><td width="<HTMLWIDTH>-<HTMLHEIGHT>*2" align="center">

HtmlConvertTarget = <pic-f1> <img src="file:///<CURRENTDIR>/pic05.png" height="150" align="top"></td><td width="<HTMLWIDTH>-<HTMLHEIGHT>*2" align="center">

# HtmlConvertToCurrentDirTarget
# PACKTERHTML や PACKTERMSG で入力された IEコンポーネント に渡される文字列のうち、
# ここで指定された文字列を current directory を示すURI に変換します。
